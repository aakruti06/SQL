✅ Create Tables
Category Table
CREATE TABLE category (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50)
);
Expense Table
CREATE TABLE expense (
    id INT AUTO_INCREMENT PRIMARY KEY,
    amount INT,
    category_id INT,
    description VARCHAR(100),
    exp_date DATE,
    FOREIGN KEY (category_id) REFERENCES category(id)
);

def add_expense():
    amount = int(input("Enter amount: "))
    category_id = int(input("Enter category id: "))
    desc = input("Enter description: ")
    date = input("Enter date (YYYY-MM-DD): ")

    query = """
    INSERT INTO expense(amount, category_id, description, exp_date)
    VALUES (%s, %s, %s, %s)
    """

    cursor.execute(query, (amount, category_id, desc, date))
    conn.commit()

    print("Expense added")
📄 VIEW ALL (JOIN)
def view_expenses():
    query = """
    SELECT e.id, e.amount, c.name, e.description, e.exp_date
    FROM expense e
    JOIN category c
    ON e.category_id = c.id
    ORDER BY e.exp_date DESC
    """

    cursor.execute(query)

    for row in cursor.fetchall():
        print(row)
🔍 FILTER BY CATEGORY
def filter_category():
    cat = input("Enter category name: ")

    query = """
    SELECT e.amount, e.description, e.exp_date
    FROM expense e
    JOIN category c
    ON e.category_id = c.id
    WHERE c.name = %s
    """

    cursor.execute(query, (cat,))

    for row in cursor.fetchall():
        print(row)
📊 MONTHLY TOTAL
def monthly_total():
    month = input("Enter month (YYYY-MM): ")

    query = """
    SELECT SUM(amount)
    FROM expense
    WHERE exp_date LIKE %s
    """

    cursor.execute(query, (month + "%",))

    print("Total:", cursor.fetchone()[0])
📈 CATEGORY-WISE SPENDING
def category_summary():
    query = """
    SELECT c.name, SUM(e.amount)
    FROM expense e
    JOIN category c
    ON e.category_id = c.id
    GROUP BY c.name
    """

    cursor.execute(query)

    for row in cursor.fetchall():
        print(row)
